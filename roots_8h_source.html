123  inline real root_itp(
124  RealFunction f, real a, real b, real tol = OPTIMIZATION_TOL,
125  unsigned int n0 = 1, real k1 = 0.0) {
126 
127  // Default value for k1
128  if (k1 == 0.0)
129  k1 = 0.2 / (b - a);
130 
131  if(a > b) {
132  TH_MATH_ERROR("root_itp", a, INVALID_ARGUMENT);
133  return nan();
134  }
135 
136  real y_a = f(a);
137  real y_b = f(b);
138 
139  if(y_a * y_b >= 0) {
140  TH_MATH_ERROR("root_itp", y_a * y_b, INVALID_ARGUMENT);
141  return nan();
142  }
143 
144  // Monotonicity of the function
145  const int monotone = (y_a < y_b) ? 1 : -1;
146 
147  real x_t, x_new;
148 
149  TH_DEBUG((b - a) / tol);
150 
151  const long double log2n = th::log2((b - a) / tol);
152 
153  TH_DEBUG(log2n);
154 
155  const long int n_half = th::floor(log2n);
156  const long int n_max = n_half + n0;
157 
158  TH_DEBUG(n_half);
159  TH_DEBUG(n_max);
160 
161  real eps = tol * pow(2.0, n_max);
162  long int iter = 0;
163 
164  while((b - a) > (2 * tol) && iter <= n_max) {
165 
166 
167  // Interpolation
168  const real x_f = (a * y_b - b * y_a) / (y_b - y_a);
169  const real x_half = 0.5 * (a + b);
170 
171 
172  // Truncation
173  const int sigma = sgn(x_half - x_f);
174  const real delta = k1 * square(b - a);
175 
176  if (delta <= abs(x_half - x_f))
177  x_t = x_f + sigma * delta;
178  else
179  x_t = x_half;
180 
181 
182  // Projection
183  const real r = eps - (b - a) / 2.0;
184 
185  if (abs(x_t - x_half) <= r)
186  x_new = x_t;
187  else
188  x_new = x_half - sigma * r;
189 
190 
191  // Update
192  const real y_new = f(x_new);
193 
194  if (monotone * y_new > 0) {
195  b = x_new;
196  y_b = y_new;
197  } else if (monotone * y_new < 0) {
198  a = x_new;
199  y_a = y_new;
200  } else {
201  return x_new;
202  }
203 
204  TH_DEBUG((b - a) / 2.0);
205 
206  eps *= 0.5;
207  iter++;
208  }
209 
210  if(abs(b - a) > 2 * tol) {
211  TH_MATH_ERROR("root_itp", (a + b) / 2.0, NO_ALGO_CONVERGENCE);
212  return nan();
213  }
214 
215  return (a + b) / 2.0;
216  }
