593  template<typename Field>
594  inline std::vector<Field> roots(
595  const polynomial<Field>& p,
596  real tolerance = OPTIMIZATION_TOL,
597  unsigned int steps = 0) {
598 
599  // Effective order of the polynomial
600  const unsigned int n = p.find_order();
601  p /= p.coeff[n];
602 
603  // Absolute value of the highest coefficient
604  Field a_hi = abs(p.coeff[n]);
605  Field a_sum = 0;
606 
607  // Sum the absolute values of the lesser coefficients
608  for (unsigned int i = 0; i < n; ++i)
609  a_sum += abs(p.coeff[i]);
610 
611  // The roots are bounded in absolute value by the maximum
612  const Field M = max(a_hi, a_sum);
613 
614  // Back track from the bounds to the first sign inversion ?
615 
616  return roots(
617  [p](real x) { return p(x); },
618  -M, M, tolerance, steps ? steps : (2 * n));
619  }
