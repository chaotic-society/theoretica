459  inline std::vector<real> roots(
460  RealFunction f, real a, real b,
461  real tolerance = BISECTION_APPROX_TOL, real steps = 10) {
462 
463  if(steps == 0) {
464  TH_MATH_ERROR("roots", steps, DIV_BY_ZERO);
465  return {nan()};
466  }
467 
468  // Find candidate intervals
469  std::vector<vec2> intervals = find_root_intervals(f, a, b, steps);
470 
471  std::vector<real> res;
472  res.reserve(intervals.size());
473 
474  // Iterate through all candidate intervals and refine the results
475  for (unsigned int i = 0; i < intervals.size(); ++i) {
476 
477  // Check whether the extremes of the candidate intervals
478  // happen to coincide with the roots
479  if(abs(f(intervals[i][0])) <= MACH_EPSILON) {
480  res.push_back(intervals[i][0]);
481  continue;
482  }
483 
484  if(abs(f(intervals[i][1])) <= MACH_EPSILON) {
485  res.push_back(intervals[i][1]);
486  continue;
487  }
488 
489  // Approximate the roots using bisection inside each interval
490  res.push_back(
491  root_bisection(f, intervals[i][0], intervals[i][1], tolerance));
492  }

494  return res;

505  template<typename Field>
506  inline std::vector<Field> roots(
507  polynomial<Field> p, real tolerance = BISECTION_APPROX_TOL,
508  unsigned int steps = 0) {
509 
510  // Effective order of the polynomial
511  const unsigned int n = p.find_order();
512  p /= p.coeff[n];
513 
514  // Absolute value of the highest coefficient
515  Field a_hi = abs(p.coeff[n]);
516  Field a_sum = 0;
517 
518  // Sum the absolute values of the lesser coefficients
519  for (unsigned int i = 0; i < n; ++i)
520  a_sum += abs(p.coeff[i]);
521 
522  // The roots are bounded in absolute value by the maximum
523  const Field M = max(a_hi, a_sum);
524 
525  // Back track from the bounds to the first sign inversion ?
526 
527  return roots(
528  [p](real x) { return p(x); },
529  -M, M, tolerance, steps ? steps : (2 * n));
530  }
