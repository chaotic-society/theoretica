32
33 real t = invlerp(P1.get(0), P2.get(0), value);

70 // Spherical interpolation
71 template<unsigned int N>
72 inline vec<N> slerp(vec<N> P1, vec<N> P2, real t) {
73
74 // Compute (only once) the length
75 // of the input vectors
76 real P1_l = P1.length();
77 real P2_l = P2.length();
78
79 // Check whether one of the vectors is null,
80 // which would make the computation impossible
81 if(P1_l == 0 || P2_l == 0) {
82 UMATH_ERROR("slerp", P1_l ? P2_l : P1_l, IMPOSSIBLE_OPERATION);
83 return vec<N>(nan());
84 }
85
86 // Angle between P1 and P2 (from the dot product)
87 real omega = acos((P1 * P2) / (P1_l * P2_l));
88 real s = sin(omega);
89
90 // Check that the sine of the angle is not zero
91 if(s == 0) {
92 UMATH_ERROR("slerp", s, DIV_BY_ZERO);
93 return vec<N>(nan());
94 }
95
96 return (P1 * sin((1 - t) * omega) + P2 * sin(t * omega)) / s;
97 }

103 // Smoothstep interpolation
104 inline real smoothstep(real x1, real x2, real interp) {
105
106 if(x1 == x2) {
107 UMATH_ERROR("smoothstep", x1, DIV_BY_ZERO);
108 return nan();
109 }
110
111 // Clamp x between 0 and 1
112 const real x = clamp((interp - x1) / (x2 - x1), 0.0, 1.0);
113
114 // 3x^2 - 2x^3
115 return x * x * (3 - 2 * x);
116 }

119 // Smootherstep interpolation
120 inline real smootherstep(real x1, real x2, real interp) {
121
122 if(x1 == x2) {
123 UMATH_ERROR("smootherstep", x1, DIV_BY_ZERO);
124 return nan();
125 }
126
127 // Clamp x between 0 and 1
128 const real x = clamp((interp - x1) / (x2 - x1), 0.0, 1.0);
129
130 // 6x^5 - 15x^4 + 10x^3
131 return x * x * x * (x * (x * 6 - 15) + 10);
132 }
