138// Compare a function to an expected value
139template<typename T1, typename T2>
140void test_tol(T1 evaluated, T1 expected, T2 input, real tolerance = TOLERANCE, bool silent = false) {
141
142 real diff = umath::abs(evaluated - expected);
143 cum_err += diff;
144 cum_sqr_err += umath::square(diff);
145 max_err = umath::max(diff, max_err);
146 tolr_test_runs++;
147
148 if(!good_enough(evaluated, expected, tolerance)) {
149
150 if(!silent) {

152 std::cout << "\t\tExpected: " << expected << std::endl;

154 std::cout << "\t\tInput: " << input << std::endl;

156 }
157

159 curr_errors++;
160
161 } else if(!silent) {
162 std::cout << "\tTest passed with diff: " << diff << std::endl;
