93 template<typename RealFunction>
94 inline real deriv_ridders(RealFunction f, real x, real h = 0.01, unsigned int degree = 3) {
95
96 real A[degree][degree];
97
98 for (unsigned int i = 0; i < degree; ++i) {
99
100 for (unsigned int n = 0; n <= i; ++n) {
101
102 unsigned int m = i - n;
103
104 if(n == 0) {
105 A[n][m] = deriv_central(f, x, h / (1 << m));
106 } else {
107 real coeff = square(1 << n);
108 A[n][m] = (coeff * A[n - 1][m + 1] - A[n - 1][m]) / (coeff - 1);
109 }
110 }
111
112 }
113
114 return A[degree - 1][0];
