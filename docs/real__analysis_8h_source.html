40 template<typename UnsignedIntType = uint64_t>
41 inline UnsignedIntType isqrt(UnsignedIntType n) {
42
43 // Upper bound
44 UnsignedIntType upper = n + 1;
45
46 // Lower bound
47 UnsignedIntType lower = 0;
48
49 // Carry for safe long division
50 UnsignedIntType c = 0;
51
52 while(lower != upper - 1) {
53
54 // Compute carry for long division by 2
55 c = ((lower % 2 != 0) && (upper % 2 != 0)) ? 1 : 0;
56
57 // Safer division by 2 for big numbers
58 const UnsignedIntType m = (lower >> 1) + (upper >> 1) + c;
59
60 // Using division instead of multiplication avoids
61 // overflows which would remove significant bits
62 const UnsignedIntType q = n / m;
63
64 if(m > q)
65 upper = m;
66 else if(m < q)
67 lower = m;
68 else
69 return m;
70 }
71
72 return lower;
73 }

186
187 if(x < 1) {
188
189 if(x == 0)
190 return 0;
191
192 // cbrt(x) is odd
193 if(x < 0)
194 return -cbrt(-x);
195
196 // Approximate cbrt between 0 and 1
197 // The root of the inverse is the inverse of the root
198 // !!! Possible precision problems with smaller numbers
199 return 1.0 / cbrt(1.0 / x);
200 }
201
202 // Approximate cbrt(x) using Newton-Raphson
203 real y = x;
204 unsigned int i = 0;
205
206 while((cube(y) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
207 y = (y * 2.0 + x / (y * y)) / 3.0;
208 i++;
209 }

408 inline real log2(real x) {
409
410 if(x <= 0) {
411
412 if(x == 0) {
413 UMATH_ERROR("log2", x, OUT_OF_RANGE);
414 return -inf();
415 }
416
417 UMATH_ERROR("log2", x, OUT_OF_DOMAIN);
418 return nan();
419 }
420
421#ifdef THEORETICA_X86
422
423 // Approximate the binary logarithm of x by
424 // exploiting x86 Assembly instructions
425 return fyl2x(x, 1.0);
426#else
427
428 // Domain reduction to [1, +inf)
429 if(x < 1)
430 return -log2(1 / x);
431
432 // Compute the biggest power of 2
433 // so that x <= 2^i
434 int i = 0;
435 while(x > (1 << i))
436 i++;
437
438 // Domain reduction to [1, 2]
439 x /= static_cast<real>(1 << i);
440
441 // Use a polynomial approximation of log2(x)
442 // in [1, 2] to compute the logarithm of the
443 // remainder.
444 // Note: Choosing a better interval or applying
445 // some transformations on the function a better
446 // approximation might be found.
447 real lr = 0;
448
449 // Exact powers of 2 don't need further computation
450 if(abs(x - 1) > MACH_EPSILON) {
451
452 // Chebyshev interpolating polynomial of degree 8
453 lr = -3.23585 + x * (7.08614
454 + x * (-7.39388 + x * (5.6659
455 + x * (-2.90591 + x * (0.945908
456 + x * (-0.176734 + x * (0.0144404 * x)))))));
457 }
458
459 return static_cast<real>(i) + lr;
460#endif
461 }

531 constexpr inline T pow(T x, int n) {
532
533 T res = T(0);
534 T x2 = x * x;
535 int i = 1;
536
537 if(n > 0) {
538
539 res = x;
540
541 for(; i < (n / 2); i += 2)
542 res *= x2;
543
544 for(; i < n; ++i)
545 res *= x;
546
547 } else if(n < 0) {
548
549 res = 1 / x;
550
551 for(; i < (n / 2); i += 2)
552 res /= x2;
553
554 for(; i < -n; ++i)
555 res /= x;
556 } else {
557 return static_cast<T>(1.0);
558 }
559
560 return res;
561 }

571 inline real root(real x, int n) {
572
573 if(((n % 2 == 0) && (x < 0)) || (n == 0)) {
574 UMATH_ERROR("root", n, OUT_OF_DOMAIN);
575 return nan();
576 }
577
578 if(n < 0)
579 return 1.0 / root(x, -n);
580
581 if(x < 1) {
582
583 if(x == 0)
584 return 0;
585
586 // Approximate root between 0 and 1
587 // The root of the inverse is the inverse of the root
588 // !!! Possible precision problems with smaller numbers
589 return 1.0 / root(1.0 / x, n);
590 }
591
592 // Approximate n-th root using Newton-Raphson
593
594 real y = x;
595 unsigned int i = 0;
596
597 while((pow(y, n) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
598 y = (y * (n - 1) + x / pow(y, n - 1)) / (real) n;
599 i++;
600 }
601
602 return y;
603 }

639 inline real exp(real x) {
640
641 // Domain reduction to [0, +inf]
642 if(x < 0)
643 return 1.0 / exp(abs(x));
644
645 real fract_x = fract(x);
646
647#ifdef THEORETICA_X86
648
649 return pow(E, floor(x)) * exp_x86_norm(fract_x);
650
651#else
652
653 // Taylor series expansion
654 // Compute e^floor(x) * e^fract(x)
655
656 real res = 1;
657 real s_n = 1;
658
659 for (int i = 1; i < 9; ++i) {
660
661 // Recurrence formula to improve
662 // numerical stability and performance
663 s_n *= (fract_x / 2.0) / static_cast<real>(i);
664 res += s_n;
665 }
666
667 return pow(E, floor(x)) * res * res;
668
669#endif
670 }

695
696#ifdef THEORETICA_X86
697
698 #ifdef MSVC_ASM
699 __asm {
700 fld x
701 fsin
702 }
703 #else
704 asm("fsin" : "+t"(x));
705 #endif
706
707 return x;
708
709#else
710
711 // Clamp x between -2PI and 2PI
712 while(x >= 2 * PI)
713 x -= 2 * PI;
714
715 while(x <= -2 * PI)
716 x += 2 * PI;
717
718 // Domain reduction to [-PI, PI]
719 if(x > PI) {
720 x = PI - x;
721 } else if(x < -PI) {
722 x = -PI - x;
723 }
724
725 real res = 0;
726
727 // Taylor series expansion
728 // sin(x) = sum( (-1)^i * x^(2i+1) / (2i+1)! )
729
730 for (int i = 0; i < TAYLOR_ORDER; ++i) {
731 res += (i % 2 == 0 ? 1 : -1)
732 * pow(x, 2 * i + 1)
733 / static_cast<real>(fact(2 * i + 1));
734 }
735
736 return res;
737#endif
738 }
