65 // using Simpson's method
66 real approx_integral_simpson(real_function f, real a, real b, unsigned int steps = INTEGRATION_STEPS) {
67

86 // Approximate the definite integral of an arbitrary function
87 // using Romberg's method accurate to the given order
88 real approx_integral_romberg(real_function f, real a, real b, unsigned int order) {
89
90 if(order % 2 != 0) {
91 UMATH_ERROR("approx_integral_romberg", order, IMPOSSIBLE_OPERATION);
92 return nan();
93 }
94
95 unsigned int iter = order / 2;
96
97 real T[iter][iter];
98
99 T[0][0] = (f(a) + f(b)) * (b - a) / 2.0;
100
101 for (int j = 1; j < iter; ++j) {
102
103 // Composite Trapezoidal Rule
104 T[j][0] = approx_integral_trapezoid(f, a, b, 1 << j);
105
106 // Richardson extrapolation
107 for (int k = 1; k <= j; ++k) {
108 T[j][k] =
109 T[j][k - 1]
110 + (T[j][k - 1] - T[j - 1][k - 1]) / ((1 << (2 * k)) - 1);
111 }
112 }
113
114 // Return the best approximation
115 return T[iter - 1][iter - 1];
116 }

165 // Runge-Kutta integration of 4th order
166 inline void integrate(kinematic_state s, real t, real dt, accel_function accel) {
167
168 kinematic_deriv a, b, c, d;
169
170 a = eval(s, t, 0, kinematic_deriv(), accel);
171 b = eval(s, t, dt * 0.5, a, accel);
172 c = eval(s, t, dt * 0.5, b, accel);
173 d = eval(s, t, dt, c, accel);
174
175 // Approximate integration using 4 different points
176 real dxdt = 1.0f / 6.0f * (a.dx + 2.0f * (b.dx + c.dx) + d.dx);
177 real dvdt = 1.0f / 6.0f * (a.dv + 2.0f * (b.dv + c.dv) + d.dv);
178
179 // Update state
180 s.x = s.x + dxdt * dt;
181 s.v = s.v + dvdt * dt;
182 }
