158 template<typename Vector>
159 inline auto sqr_norm(const Vector& v) {
160
161 using Type = decltype(v.get(0));
162 Type sum = (Type) 0;
163
164 // Use conjugation for complex numbers
165 if(is_complex_type<Type>::value)
166 for (unsigned int i = 0; i < v.size(); ++i)
167 sum += v.get(i) * conjugate(v.get(i));
168 else
169 for (unsigned int i = 0; i < v.size(); ++i)
170 sum += v.get(i) * v.get(i);
171
172 return sum;
173 }

449 template<typename Matrix1, typename Matrix2>
450 inline Matrix1& inverse(Matrix1& dest, const Matrix2& src) {
451
452 if(src.rows() != src.cols()) {
453 TH_MATH_ERROR("algebra::inverse", src.rows(), INVALID_ARGUMENT);
454 mat_error(dest);
455 return dest;
456 }
457
458 if(dest.rows() != src.rows()) {
459 TH_MATH_ERROR("algebra::inverse", dest.rows(), INVALID_ARGUMENT);
460 mat_error(dest);
461 return dest;
462 }
463
464 if(dest.cols() != src.cols()) {
465 TH_MATH_ERROR("algebra::inverse", dest.cols(), INVALID_ARGUMENT);
466 mat_error(dest);
467 return dest;
468 }
469
470 using Type = decltype(src.get(0, 0));
471
472 // Prepare extended matrix (A|B)
473 Matrix1 A;
474 A.resize(src.rows(), src.cols());
475 dest.resize(src.rows(), src.cols());
476 mat_copy(A, src);
477 make_identity(dest);
478
479 // Iterate on all columns
480 for (unsigned int i = 0; i < src.rows(); ++i) {
481
482 // Make sure the element on the diagonal
483 // is non-zero by adding the first non-zero row
484 if(A.at(i, i) == (Type) 0) {
485
486 bool flag = false;
487
488 // Iterate on all rows
489 for (unsigned int j = i + 1; j < src.rows(); ++j) {
490
491 // Add the j-th row to the i-th row
492 // if Aji is non-zero
493 if(A.at(j, i) != (Type) 0) {
494
495 for (unsigned int k = 0; k < src.rows(); ++k) {
496 A.at(i, k) += A.at(j, k);
497 dest.at(i, k) += dest.at(j, k);
498 }
499
500 flag = true;
501 break;
502 }
503 }
504
505 if(!flag) {
506 TH_MATH_ERROR("algebra::inverse", flag, IMPOSSIBLE_OPERATION);
507 mat_error(dest);
508 return dest;
509 }
510 }
511
512 auto inv_pivot = ((Type) 1.0) / A.at(i, i);
513
514 // Use the current row to make all other
515 // elements of the column equal to zero
516 for (unsigned int j = 0; j < src.rows(); ++j) {
517
518 // Skip the current row
519 if(j == i)
520 continue;
521
522 // Multiplication coefficient for
523 // the elision of Ajk
524 const auto coeff = A.at(j, i) * inv_pivot;
525
526 for (unsigned int k = 0; k < src.rows(); ++k) {
527 A.at(j, k) -= coeff * A.at(i, k);
528 dest.at(j, k) -= coeff * dest.at(i, k);
529 }
530 }
531
532 // Divide the current row by the pivot
533 for (unsigned int j = 0; j < src.rows(); ++j) {
534 A.at(i, j) *= inv_pivot;
535 dest.at(i, j) *= inv_pivot;
536 }
537
538 }
539
540 return dest;
541 }

621 template<typename Matrix>
622 inline auto det(const Matrix& m) {
623
624 using Type = decltype(m.get(0, 0));
625
626 if(m.rows() != m.cols()) {
627 TH_MATH_ERROR("algebra::det", m.rows(), INVALID_ARGUMENT);
628 return (Type) nan();
629 }
630
631 Matrix A;
632 A.resize(m.rows(), m.cols());
633 mat_copy(A, m);
634
635 // Iterate on all columns
636 for (unsigned int i = 0; i < A.rows(); ++i) {
637
638 // Make sure the element on the diagonal
639 // is non-zero by adding the first non-zero row
640 if(A.at(i, i) == 0) {
641
642 bool flag = false;
643
644 // Iterate on all rows
645 for (unsigned int j = i + 1; j < A.rows(); ++j) {
646
647 // Add the j-th row to the i-th row
648 // if Aji is non-zero.
649 // The determinant does not change
650 // when adding a row to another one
651 if(A.at(j, i) != (Type) 0) {
652
653 for (unsigned int k = 0; k < A.rows(); ++k) {
654 A.at(i, k) += (Type) A.at(j, k);
655 }
656
657 flag = true;
658 break;
659 }
660 }
661
662 if(!flag) {
663 return (Type) 0;
664 }
665 }
666
667 const auto inv_pivot = ((Type) 1.0) / A.at(i, i);
668
669 // Use the current row to make all other
670 // elements of the column equal to zero
671 for (unsigned int j = i + 1; j < A.rows(); ++j) {
672
673 // Multiplication coefficient for
674 // the elision of Ajk
675 const auto coeff = (Type) A.at(j, i) * inv_pivot;
676
677 // The coefficient does not change
678 // when adding a linear combination
679 // of a row to another
680 for (unsigned int k = 0; k < A.rows(); ++k) {
681 A.at(j, k) -= (Type) A.at(i, k) * coeff;
682 }
683 }
684 }
685
686 // The determinant of a (lower) triangular matrix
687 // is the product of the elements on its diagonal
688 return (Type) diagonal_product(A);
689 }

1517 m.at(2, 1) = Rz * Ry * cm1 + Rx * s;
1518 m.at(2, 2) = c + Rz * Rz * cm1;

1758 template<
1759 typename Matrix, typename Vector1,
1760 typename Vector2, typename Vector3>
1761 inline Matrix look_at(Vector1 camera, Vector2 target, Vector3 up) {
1762
1763 // Construct an orthonormal basis
1764
1765 Vector1 x_axis, y_axis, z_axis;
1766 x_axis.resize(3);
1767 y_axis.resize(3);
1768 z_axis.resize(3);
1769
1770 // z = target - camera
1771 vec_diff(z_axis, target, camera);
1772 make_normalized(z_axis);
1773
1774 // x = z X up
1775 vec_copy(x_axis, cross(z_axis, up));
1776 make_normalized(x_axis);
1777
1778 // y = x X z
1779 vec_copy(y_axis, cross(x_axis, z_axis));
1780
1781 // Negate z_axis to have a right-handed system
1782 vec_scalmul(-1.0, z_axis);
1783
1784 // Construct the rotation and translation matrix
1785 Matrix m;
1786 m.resize(4, 4);
1787
1788 m.at(0, 0) = x_axis.get(1);
1789 m.at(0, 1) = x_axis.get(2);
1790 m.at(0, 2) = x_axis.get(3);
1791 m.at(0, 3) = dot(camera, vec_scalmul(-1.0, x_axis));
1792
1793 m.at(1, 0) = y_axis.get(1);
1794 m.at(1, 1) = y_axis.get(2);
1795 m.at(1, 2) = y_axis.get(3);
1796 m.at(1, 3) = dot(camera, vec_scalmul(-1.0, y_axis));
1797
1798 m.at(2, 0) = z_axis.get(1);
1799 m.at(2, 1) = z_axis.get(2);
1800 m.at(2, 2) = z_axis.get(3);
1801 m.at(2, 3) = dot(camera, vec_scalmul(-1.0, z_axis));
1802
1803 m.at(3, 0) = 0;
