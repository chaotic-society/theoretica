117 template <
118 typename Function, typename Vector = vec<real>,
119 enable_scalar_field<Function> = true,
120 enable_vector<Vector> = true
121 >
122 inline auto gradient(Function f, const Vector& x) {
123
124 // Extract the return type of the function
125 using MultidualType = return_type_t<Function>;
126
127 constexpr size_t N = MultidualType::size_argument;
128 vec<MultidualType, N> arg;
129 arg.resize(x.size());
130
131 // Iterate over each element, setting the dual part to
132 // the i-th element of the canonical base
133 for (unsigned int i = 0; i < x.size(); ++i) {
134 arg[i] = MultidualType(
135 x[i], vec<real, N>::euclidean_base(i, x.size())
136 );
137 }
138
139 return f(arg).Dual();
140 }

229 template <
230 typename MultidualFunction, typename Vector,
231 enable_vector<Vector> = true,
232 enable_vector_field<MultidualFunction> = true
233 >
234 inline auto jacobian(MultidualFunction f, const Vector& x) {
235
236 constexpr size_t M = return_type_t<MultidualFunction>::size_argument;
237 constexpr size_t N = _internal::func_helper<MultidualFunction>::first_arg_type::size_argument;
238
239 const vec<multidual<N>, N> res = f(multidual<N>::make_argument(x));
240
241 // Construct the jacobian matrix
242 mat<real, M, N> J;
243 J.resize(res.size(), x.size());
244
245 for (unsigned int j = 0; j < J.rows(); ++j)
246 for (unsigned int i = 0; i < res[j].v.size(); ++i)
247 J(j, i) = res[j].v[i];
248
249 return J;
250 }

283 template <
284 typename MultidualFunction, typename Vector,
285 enable_vector<Vector> = true,
286 enable_vector_field<MultidualFunction> = true
287 >
288 inline auto curl(MultidualFunction f, const Vector& x) {
289
290 Vector res;
291 res.resize(3);
292
293 if(x.size() != 3) {
294 TH_MATH_ERROR("th::curl", x.size(), MathError::InvalidArgument);
295 return algebra::vec_error(res);
296 }
297
298 constexpr size_t N = return_type_t<MultidualFunction>::size_argument;
299 const mat<real, N, N> J = jacobian(f, x);
300
301 res[0] = J(2, 1) - J(1, 2);
302 res[1] = J(0, 2) - J(2, 0);
303 res[2] = J(1, 0) - J(0, 1);
304
305 return res;
306 }

340 template <
341 typename Dual2Function, typename Vector,
342 enable_vector<Vector> = true
343 >
344 inline real laplacian(Dual2Function f, const Vector& x) {
345
346 // Extract size template argument of the input vector
347 constexpr size_t N = _internal::func_helper<Dual2Function>
348 ::first_arg_type::size_argument;
349
350 vec<dual2, N> d;
351 d.resize(x.size());
352
353 for (unsigned int i = 0; i < d.size(); ++i)
354 d[i].a = x[i];
355
356 // Accumulate second derivatives
357 real res = 0;
358 for (unsigned int i = 0; i < d.size(); ++i) {
359 d[i].b = 1;
360 res += f(d).Dual2();
361 d[i].b = 0;
362 }
363
364 return res;
365 }
