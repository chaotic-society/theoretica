117 template <
118 typename Function, typename Vector = vec<real>,
119 enable_scalar_field<Function> = true,
120 enable_vector<Vector> = true
121 >
122 inline auto gradient(Function f, const Vector& x) {
123
124 // Extract the return type of the function
125 using MultidualType = return_type_t<Function>;
126
127 constexpr size_t N = MultidualType::size_argument;
128 vec<MultidualType, N> arg;
129 arg.resize(x.size());
130
131 // Iterate over each element, setting the dual part to
132 // the i-th element of the canonical base
133 for (unsigned int i = 0; i < x.size(); ++i) {
134 arg[i] = MultidualType(
135 x[i], vec<real, N>::euclidean_base(i, x.size())
136 );
137 }
138
139 return f(arg).Dual();
140 }

179 template <
180 typename Function, typename Vector = vec<real>,
181 enable_vector_field<Function> = true,
182 enable_vector<Vector> = true
183 >
184 inline real divergence(Function V, const Vector& x) {
185
186 using MultidualT = return_type_t<Function>;
187 const size_t N = MultidualT::size_argument;
188
189 // Sum the diagonal elements of the jacobian
190 const vec<multidual<N>, N> res = V(multidual<N>::make_argument(x));
191
192 real div = 0.0;
193 for (unsigned int i = 0; i < res.size(); ++i) {
194
195 // dreal numbers initialized to a scalar
196 // have an empty dual part, check the size before summing
197 if (res[i].v.size() > i)
198 div += res[i].Dual()[i];
199 }
200
201 return div;
202 }

237 template <
238 typename MultidualFunction, typename Vector,
239 enable_vector<Vector> = true,
240 enable_vector_field<MultidualFunction> = true
241 >
242 inline auto jacobian(MultidualFunction f, const Vector& x) {
243
244 constexpr size_t M = return_type_t<MultidualFunction>::size_argument;
245 constexpr size_t N = _internal::func_helper<MultidualFunction>::first_arg_type::size_argument;
246
247 const vec<multidual<N>, N> res = f(multidual<N>::make_argument(x));
248
249 // Construct the jacobian matrix
250 mat<real, M, N> J;
251 J.resize(res.size(), x.size());
252
253 for (unsigned int j = 0; j < J.rows(); ++j)
254 for (unsigned int i = 0; i < res[j].v.size(); ++i)
255 J(j, i) = res[j].v[i];
256
257 return J;
258 }

291 template <
292 typename MultidualFunction, typename Vector,
293 enable_vector<Vector> = true,
294 enable_vector_field<MultidualFunction> = true
295 >
296 inline auto curl(MultidualFunction f, const Vector& x) {
297
298 Vector res;
299 res.resize(3);
300
301 if(x.size() != 3) {
302 TH_MATH_ERROR("th::curl", x.size(), MathError::InvalidArgument);
303 return algebra::vec_error(res);
304 }
305
306 constexpr size_t N = return_type_t<MultidualFunction>::size_argument;
307 const mat<real, N, N> J = jacobian(f, x);
308
309 res[0] = J(2, 1) - J(1, 2);
310 res[1] = J(0, 2) - J(2, 0);
311 res[2] = J(1, 0) - J(0, 1);
312
313 return res;
314 }

348 template <
349 typename Dual2Function, typename Vector,
350 enable_vector<Vector> = true
351 >
352 inline real laplacian(Dual2Function f, const Vector& x) {
353
354 // Extract size template argument of the input vector
355 constexpr size_t N = _internal::func_helper<Dual2Function>
356 ::first_arg_type::size_argument;
357
358 vec<dual2, N> d;
359 d.resize(x.size());
360
361 for (unsigned int i = 0; i < d.size(); ++i)
362 d[i].a = x[i];
363
364 // Accumulate second derivatives
365 real res = 0;
366 for (unsigned int i = 0; i < d.size(); ++i) {
367 d[i].b = 1;
368 res += f(d).Dual2();
369 d[i].b = 0;
370 }
371
372 return res;
373 }
