1599  inline mat<Type>& resize(unsigned int n, unsigned int k) {
1600 
1601  // Do nothing if the size is already correct
1602  if(rows() == n && cols() == k)
1603  return *this;
1604 
1605  // Distinguish between row-first and column-first allocation
1606 #ifdef THEORETICA_ROW_FIRST
1607  size_t size1 = n, size2 = k;
1608 #else
1609  size_t size1 = k, size2 = n;
1610 #endif
1611 
1612  // The matrix must be allocated anew
1613  if(!data.size()) {
1614 
1615  data.resize(size1);
1616  for (unsigned int i = 0; i < size1; ++i)
1617  data[i].resize(size2);
1618 
1619  row_sz = n;
1620  col_sz = k;
1621  algebra::mat_zeroes(*this);
1622 
1623  // The matrix must be reallocated
1624  } else {
1625 
1626  std::vector<std::vector<Type>> new_data;
1627  new_data.resize(size1);
1628  for (unsigned int i = 0; i < size1; ++i)
1629  new_data[i].resize(size2);
1630 
1631  // Copy data to new memory location
1632 
1633  // Bounds on the region to copy
1634  const size_t row_bound = min(rows(), n);
1635  const size_t col_bound = min(cols(), n);
1636 
1637  for (unsigned int i = 0; i < row_bound; ++i) {
1638  for (unsigned int j = 0; j < col_bound; ++j) {
1639 #ifdef THEORETICA_ROW_FIRST
1640  new_data[i][j] = get(i, j);
1641 #else
1642  new_data[j][i] = get(i, j);
1643 #endif
1644  }
1645  }
1646 
1647  // If the new matrix size is bigger,
1648  // zero out all remaining entries
1649  for (unsigned int i = 0; i < n; ++i) {
1650  for (unsigned int j = col_bound; j < k; ++j) {
1651 
1652 
1653 #ifdef THEORETICA_ROW_FIRST
1654  new_data[i][j] = (Type) 0;
1655 #else
1656  new_data[j][i] = (Type) 0;
1657 #endif
1658  }
1659  }
1660 
1661 
1662  // If the new matrix size is bigger,
1663  // zero out all remaining entries
1664  for (unsigned int i = row_bound; i < n; ++i) {
1665  for (unsigned int j = 0; j < k; ++j) {
1666 
1667 
1668 #ifdef THEORETICA_ROW_FIRST
1669  new_data[i][j] = (Type) 0;
1670 #else
1671  new_data[j][i] = (Type) 0;
1672 #endif
1673  }
1674  }
1675 
1676  data.clear();
1677  data = new_data;
1678 
1679  row_sz = n;
1680  col_sz = k;
1681  }
1682 
1683  return *this;
1684  }
