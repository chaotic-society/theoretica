465  inline real pvalue_chi_squared(real chi_sqr, unsigned int ndf) {
466 
467  if(ndf == 0) {
468  TH_MATH_ERROR("pvalue_chi_squared", ndf, INVALID_ARGUMENT);
469  return nan();
470  }
471 
472  // For ndf >= 260 use the Gaussian approximation
473  // as the coefficients are not stable
474  if(ndf >= 260) {
475 
476  const real new_x = (chi_sqr - ndf) / sqrt(2.0 * ndf);
477 
478  // For really low Chi-squared the Gaussian is
479  // below tolerance value for integration
480  if(new_x < 0) {
481 
482  if(new_x < -3)
483  return 1 - integral_inf_riemann([=](real x) {
484  return exp(-x * x / 2) / SQRTPI / SQRT2;
485  }, -new_x, 1E-16, 25);
486 
487  return 0.5 + integral_romberg_tol([=](real x) {
488  return exp(-x * x / 2) / SQRTPI / SQRT2;
489  }, new_x, 0, 1E-16);
490  } else {
491 
492  if(new_x > 3)
493  return integral_inf_riemann([=](real x) {
494  return exp(-x * x / 2) / SQRTPI / SQRT2;
495  }, new_x, 1E-16, 25);
496 
497  return 0.5 - integral_romberg_tol([=](real x) {
498  return exp(-x * x / 2) / SQRTPI / SQRT2;
499  }, 0, new_x, 1E-16);
500  }
501  }
502 
503  // Compute the coefficient using a stable equivalent formula
504  const real coeff = exp(-special::lngamma(ndf / 2.0) - chi_sqr / 2.0);
505 
506  // Use different methods when Gauss-Laguerre is not numerically stable
507  if((ndf > 70 && chi_sqr < (ndf / 2.0))) {
508 
509  // Use equivalent formula around potential singularity
510  real res = integral_romberg_tol([=](real x) {
511  return pow(sqrt(x + chi_sqr / 2), ndf - 2) * exp(-x);
512  }, 0, 1, 1E-12);
513 
514  res += integral_inf_riemann([=](real x) {
515  return exp((ndf - 2) / 2.0 * ln(x + chi_sqr / 2) - x);
516  }, 1, ndf / 2, 1E-12, 25);
517 
518  return coeff * res;
519  }
520 
521  // Approximate the integral using Gauss-Laguerre quadrature
522  return coeff * integral_gauss(
523  [=](real x) {
524  return pow(sqrt(x + chi_sqr / 2), ndf - 2);
525  }, tables::laguerre_roots_16, tables::laguerre_weights_16, 16);
526  }
