553  inline real pvalue_chi_squared(real chi_sqr, unsigned int ndf) {
554 
555  if(ndf == 0) {
556  TH_MATH_ERROR("pvalue_chi_squared", ndf, INVALID_ARGUMENT);
557  return nan();
558  }
559 
560  // For ndf >= 260 use the Gaussian approximation
561  // as the coefficients are not stable
562  if(ndf >= 260) {
563 
564  const real new_x = (chi_sqr - ndf) / sqrt(2.0 * ndf);
565 
566  // For really low Chi-squared the Gaussian is
567  // below tolerance value for integration
568  if(new_x < 0) {
569 
570  if(new_x < -3)
571  return 1 - integral_inf_riemann([=](real x) {
572  return exp(-x * x / 2) / SQRTPI / SQRT2;
573  }, -new_x, 1E-16, 25);
574 
575  return 0.5 + integral_romberg_tol([=](real x) {
576  return exp(-x * x / 2) / SQRTPI / SQRT2;
577  }, new_x, 0, 1E-16);
578  } else {
579 
580  if(new_x > 3)
581  return integral_inf_riemann([=](real x) {
582  return exp(-x * x / 2) / SQRTPI / SQRT2;
583  }, new_x, 1E-16, 25);
584 
585  return 0.5 - integral_romberg_tol([=](real x) {
586  return exp(-x * x / 2) / SQRTPI / SQRT2;
587  }, 0, new_x, 1E-16);
588  }
589  }
590 
591  // Compute the coefficient using a stable equivalent formula
592  const real coeff = exp(-special::lngamma(ndf / 2.0) - chi_sqr / 2.0);
593 
594  // Use different methods when Gauss-Laguerre is not numerically stable
595  if((ndf > 70 && chi_sqr < (ndf / 2.0))) {
596 
597  // Use equivalent formula around potential singularity
598  real res = integral_romberg_tol([=](real x) {
599  return pow(sqrt(x + chi_sqr / 2), ndf - 2) * exp(-x);
600  }, 0, 1, 1E-12);
601 
602  res += integral_inf_riemann([=](real x) {
603  return exp((ndf - 2) / 2.0 * ln(x + chi_sqr / 2) - x);
604  }, 1, ndf / 2, 1E-12, 25);
605 
606  return coeff * res;
607  }
608 
609  // Approximate the integral using Gauss-Laguerre quadrature
610  return coeff * integral_gauss(
611  [=](real x) {
612  return pow(sqrt(x + chi_sqr / 2), ndf - 2);
613  }, tables::laguerre_roots_16, tables::laguerre_weights_16, 16);
614  }
