227 inline polynomial operator/(const polynomial& d) const {
228
229 const unsigned int d_order = d.degree();
230 const unsigned int this_order = degree();
231
232 if(d_order == 0 && d[0] == 0) {
233 TH_MATH_ERROR("polynomial::operator/", d[0], MathError::DivByZero);
234 return polynomial(nan());
235 }
236
237 // Remainder
238 polynomial r = *this;
239
240 // Quotient
241 polynomial q = polynomial();
242 unsigned int i = 0;
243
244 while(i <= this_order) {
245
246 // Compute only once the degree of the polynomial
247 const unsigned int r_order = r.degree();
248
249 // Stop execution if the division is complete
250 // (when the remainder is 0 or has lower degree)
251 if((r_order == 0 && (abs(r[0]) < MACH_EPSILON)) || r_order < d_order)
252 break;
253
254 // Simple division between highest degree terms
255 const polynomial t = polynomial<Type>::monomial(
256 r[r_order] / d[d_order],
257 r_order - d_order);
258
259 // Add monomial to quotient and subtract the
260 // monomial times the dividend from the remainder
261 q += t;
262 r -= t * d;
263
264 i++;
265 }
266
267 // Check if division failed to converge
268 if(i > this_order && r.degree() >= d_order) {
269 TH_MATH_ERROR("polynomial::operator/", i, MathError::NoConvergence);
270 return polynomial(nan());
271 }
272
273 return q;
274 }

422 inline vec<complex<>, 2> quadratic_roots() const {
423
424 const int order = degree();
425
426 // Check that the polynomial is quadratic
427 if(order != 2) {
428 TH_MATH_ERROR("quadratic_roots", order, MathError::ImpossibleOperation);
429 return vec<complex<>, 2>({nan(), nan()});
430 }
431
432 complex<> z1, z2;
433 const Type a = coeff[2];
434 const Type b = coeff[1];
435 const Type c = coeff[0];
436
437 // Linear case
438 if (abs(a) < MACH_EPSILON) {
439
440 if (abs(b) < MACH_EPSILON)
441 return {nan(), nan()};
442
443 return {-c/b, -c/b};
444 }
445
446 const Type discriminant = b * b - 4 * a * c;
447 const complex<> delta = sqrt(complex<>(discriminant));
448
449 const real sign = (b > 0) ? 1 : -1;
450 const complex<> q = -0.5 * (b + sign * delta);
451
452 if (abs(q) < MACH_EPSILON)
453 return {0.0, 0.0};
454
455 z1 = q / a;
456 z2 = c / q;
457
458 return {z1, z2};
459 }
