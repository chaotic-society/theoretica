137  template<typename Vector>
138  inline real sum_compensated(const Vector& X) {
139 
140  // Total sum
141  real sum = 0;
142 
143  // Correction term
144  real corr = 0;
145 
146  for (unsigned int i = 0; i < X.size(); i++) {
147 
148  const real temp = sum + X[i];
149 
150  // Sort the two addends to preserve bits
151  corr += (abs(sum) >= abs(X[i]))
152  ? ((sum - temp) + X[i])
153  : ((X[i] - temp) + sum);
154 
155  sum = temp;
156  }
157 
158  // Apply correction
159  return sum + corr;
160  }

172  inline real sum_pairwise(
173  const Vector& X, size_t begin = 0,
174  size_t end = 0, size_t base_size = 128) {
175 
176  if(end == 0)
