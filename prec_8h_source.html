184  inline void estimate(
185  const std::string& name,
186  Function1 funcApprox,
187  Function2 funcExpected,
188  estimate_options<R, Args...> opt) {
189 
190  // Skip the test case if any tests have been picked
191  // and this one was not picked.
192  if(state.pickedTests.size())
193  if(state.pickedTests.find(name) == state.pickedTests.end())
194  return;
195 
196  // Use the estimator to estimate error integrals.
197  auto res = opt.estimator(funcApprox, funcExpected, opt);
198 
199  res.funcName = name;
200  res.domain = opt.domain;
201  res.tolerance = opt.tolerance;
202  res.quiet = opt.quiet;
203  res.iterations = opt.iterations;
204 
205  // Use the fail function to determine whether the test failed.
206  res.failed = opt.fail(res);
207 
208  state.totalTests++;
209  if(res.failed)
210  state.failedTests++;
211 
212  state.estimateResults[name].push_back(res);
213  }
