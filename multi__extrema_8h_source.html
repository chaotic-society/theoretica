118 inline ReturnVector multi_minimize_lingrad(
119 DualObjectiveFunction f,
120 Vector guess,
121 real tolerance = OPTIMIZATION_MINGRAD_TOLERANCE,
122 unsigned int max_iter = OPTIMIZATION_MINGRAD_ITER
123 ) {
124
125 ReturnVector x = guess;
126 ReturnVector grad;
127 unsigned int iter = 0;
128
129 constexpr size_t N = ReturnVector::size_argument;
130
131 do {
132
133 grad = autodiff::gradient(f, x);
134
135 // Minimize f(x + gamma * gradient) in [-1, 0]
136 // using Golden Section extrema search
137 real gamma = minimize_golden(
138 [f, x, grad](real gamma){
139 return f(multidual<N>::make_argument(x + gamma * grad)).Re();
140 }, -1, 0);
141
142 // Fallback value
143 if(-gamma <= MACH_EPSILON)
144 gamma = OPTIMIZATION_MINGRAD_GAMMA;
145
146 x += gamma * grad;
147 iter++;
148
149 } while(grad.norm() > tolerance && iter <= max_iter);
150
151 if(iter > max_iter) {
152 TH_MATH_ERROR("multi_minimize_lingrad", iter, MathError::NoConvergence);
153 return algebra::vec_error(x);
154 }
155
156 return x;
157 }
